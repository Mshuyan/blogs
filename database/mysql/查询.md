select distinct ir_sitename from URLCONTENT;
统计这一列 去重后的数量
EXPLAIN select count(distinct ir_sitename) from URLCONTENT;
explain SELECT count(ir_sitename) FROM URLCONTENT;
统计这一列中，不同的值的数量
SELECT IR_SITENAME ,count(*) num FROM URLCONTENT GROUP BY IR_SITENAME;
